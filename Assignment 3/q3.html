def mylinridgereg(X, Y, ):
    X = np.hstack((np.ones((X.shape[0], 1)), X))
    return np.dot(np.linalg.inv(np.dot(X.T, X) + np.dot(( * np.eye(X.shape[1])).T,  * np.eye(X.shape[1]))), np.dot(X.T, Y))

def mylinridgeregeval(X, weights):
    X = np.hstack((np.ones((X.shape[0], 1)), X))
    return np.dot(X, weights)
